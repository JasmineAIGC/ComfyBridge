日志
使用concurrent_log_handler作为logging的handle。
例子：
import time
import os
# import logging
# from loguru import logger
import logging.config
from pathlib import Path

configured = False

def loggers(DEBUG=False, MAX_LOG_FILES=10, SHOW=True):
    global configured
    if not configured:
        # if not SHOW:
        #     logger.remove()
        current_dir = os.path.dirname(os.path.abspath(__file__))
        
        # 生成时间戳和随机后缀
        timestamp = time.strftime("%Y%m%d%H%M%S")
        # 构造日志文件路径
        log_dir = f'{current_dir}/logs/'
        logs_list = Path(log_dir).glob('*.log')
        logs_list = sorted(logs_list)
        log_dir = "/home/finance/Logs"
        for index in range(len(logs_list) - MAX_LOG_FILES):
            os.remove(logs_list[index])
        log_file_path = f'{log_dir}/robot-tts-cosyvoicev2_{timestamp}.log'
        if not os.path.exists(os.path.dirname(log_file_path)):
            os.makedirs(os.path.dirname(log_file_path))  # 如果目录不存在，创建目录
        if not os.access(os.path.dirname(log_file_path), os.W_OK):
            print("Error: No write permission for the log file path.")
            exit(1)
        if DEBUG:
            log_level = 'DEBUG'
        else:
            log_level = 'INFO'
 
        LOG_HANDLER = "concurrent_log_handler.ConcurrentTimedRotatingFileHandler"
        # 日志相关配置
        logging.config.dictConfig({
            'version': 1,
            'disable_existing_loggers': False,  # 防止在模块层次创建了logger而使自定义的logger失效
            'formatters': {  # 设置日志输出的基本格式
                'standard': {  # 日志格式对象
                    'format': '%(levelname)s %(asctime)s %(name)s %(filename)s#%(lineno)d: %(message)s',
                    'datefmt': '%Y-%m-%d %H:%M:%S',
                    'encoding': "utf-8",
                },
            },
            'handlers': {  # 配置日志的处理器对象
                'service': {  # 日志处理器对象的名称
                    'level': log_level,  # 日志处理其对象的等级
                    'class': LOG_HANDLER,  # 日志处理器对象所属的类
                    'when': 'H',
                    'interval': 12,
                    'backupCount': 60,
                    'maxBytes': 512*1024*1024,
                    'delay': True,
                    'formatter': 'standard',  # 指定日志格式对象
                    'filename': log_file_path,  # 日志处理器对象写入的日志路径
                },
            },
            'loggers': {
                'service': {  # 日志对象
                    'handlers': ['service'],  # 指定日志处理器对象
                    'level': log_level,  # 日志等级
                    'propagate': True,  # 子级logger是否向父级logger传递
                }
            },
        })
        # logger.add(log_file_path, 
        #         retention=5,  
        #         backtrace=True,  # 启用堆栈跟踪
        #         enqueue=True,  # 启用异步日志记录
        #         compression='zip', 
        #         rotation="500 MB",
        #         level=log_level, 
        #         encoding="utf-8")
        configured = True
        logger = logging.getLogger("service")
    return logger
